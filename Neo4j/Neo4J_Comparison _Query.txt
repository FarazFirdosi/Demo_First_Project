Graph Query No.1
match (m:Movie) return m.released as year, count(*) as total order by year desc;
Graph Query No.2 Identify how often actors and directors worked together
MATCH (actor:Person)-[:ACTED_IN]->(movie:Movie)<-[:DIRECTED]-(director:Person)
RETURN actor,director,count(*) AS collaborations
Graph Query No. 3
MATCH (a)-[:ACTED_IN]->(m)<-[:DIRECTED]-(d) RETURN a,m,d LIMIT 10;
Graph Query No.4
match (m)-[r]->(n) 
return m.name, type(r),n.title, n.released
order by n.released desc;
Graph No.5

Graph No.6
